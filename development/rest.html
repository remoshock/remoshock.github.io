/remoshock/command

This REST service performances an action, for example it will triggering a shock. 

async function command(token, receiver, action, power, duration) {
	let url = "/remoshock/command"
            + "?receiver=" + escape(receiver)
            + "&action=" + escape(action)
            + "&power=" + escape(power)
            + "&duration=" + escape(duration);
	return fetch(url, {
		method: "POST",
		headers: {
			Authorization: "bearer " + token
		}
	});
}

await command(token, 1, "shock", 5, 250);
